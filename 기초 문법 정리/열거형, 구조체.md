### ✔️ 열거형
- 열거형은 **숫자에 별명을 붙여** 알아보기 쉽게 만드는 기능
- `int`, `float` 과 다르게 `enum` 은 사용자가 직접 만드는 자료형

```csharp
	enum GameObjectType
    {
        Monster, Item, Npc, FieldObject // 0,1,2,3
    }
    
    enum GameObjectType
    {
        Monster = 3, Item, Npc = 7, FieldObject // 3,4,7,8
    }
    
    enum GameObjectType
    {
        Monster = 3, Item, Npc = 4, FieldObject // 3,4,X,5  // Npc가 4로 중복
    }
```

- 별명으로 정의되지 않은 정수는 정수 그대로 출력.
- `int`로 형변환하면 별명이 붙어진 정수가 나타남.

```csharp
		Console.WriteLine((int)GameObjectType.Monster); // 0 출력
        Console.WriteLine((GameObjectType)1);			// Item 출력
        // enum에 정의되지 않은 값은 숫자 형태로 출력
        
 		GameObjectType type = (GameObjectType)1;  		// 정수를 형변환해서 사용도 가능
```
- 아래 함수처럼 `switch`문에서 사용될 때 
  숫자가 아닌 별명을 붙인 열거형을 이용하면 코드를 알아보기 쉬움.

```csharp
	static void OnCollision(GameObjectType type)
    {
        switch (type)
        {
            case GameObjectType.Monster:
                Console.WriteLine("몬스터와 부딪혔다. 피격처리");
                break;
            case GameObjectType.Item:
                Console.WriteLine("아이템과 부딪혔다. 먹기");
                break;
            case GameObjectType.Npc:
                Console.WriteLine("NPC와 부딪혔다. 말을 건다");
                break;
            default:
                Console.WriteLine("아무 일도 없다");
                break;
        }
    }
    
    	// Main에서 사용 시
    	GameObjectType type = GameObjectType.Monster; // 알아보기 편리함.
        OnCollision(type);     
```

### ✔️ 구조체

- 구조체는 **값 타입**임 (저장 위치는 상황에 따라 달라짐)
- .NET의 기본 자료형은 구조체로 정의되어 있음
- **데이터를 보관하는 컨테이너** 형태로 이용함

```csharp
	struct StudentData 
    // 구조체에서 변수 선언, 생성자로 외부에서 변수 초기화, 구조체 내부에서 함수 생성, 외부 출력
    {
            // 이름, 나이, 키, 지원동기
            // private은 구조체 내부에서만 접근이 가능함.
            public string Name;
            public int Age;
            public float Height;
            public string Motive;
            
            // 생성자 : 구조체의 실체가 만들어질 때 호출되는 함수
            public StudentData(string name, int age, float height, string motive)
            // 입력되는 자료형과 인자값
            {
                Name = name;
                Age = age;
                Height = height;
                Motive = motive;
            }
            
            public void PrintInfo()  // 내부에서 접근하는 함수이므로 바로 적용 가능
            {
                Console.WriteLine($"Name: {Name}");    //StudentData.Name X
                Console.WriteLine($"Age: {Age} 살");
                Console.WriteLine($"Height: {Height} cm");
                Console.WriteLine($"Motive: {Motive}");
            }
    }
```
#### 열거형과 구조체를 이용한 예시

- Item Struct
	- 내부 속성으로 이름과 고유 ID(int)를 가지며, 자신의 정보를 출력하는 기능을 가진다
- Inventory Struct
	- 내부 속성으로 아이템을 관리하는 배열과, 배열에 들어있는 아이템의 정보를 전부 출력하는 기능을 가진다
```csharp
 	enum ItemInfo 			// 열거형으로 아이템을 표현
    {
        체력포션, 마나포션, 철검, 철방어구
    }
    
    struct ItemData				// 아이템 데이터 구조체
    {
        public string Name;    // 이름, 고유 ID 변수 선언
        public int ID;

        public ItemData(ItemInfo item)    	// item 데이터를 직접 넣을 때 사용하는 생성자
        {
            Name = item.ToString();			// 문자열로 형변환하여 열거형 자체 이름을 사용
            ID = (int)item;					// 열거형의 정수형을 ID로 사용
        }

        public void PrintItemInfo()     // 아이템 이름, 고유 ID 출력 함수
        {
            Console.WriteLine($"이름 : {Name}\tID : {ID}");
        }
    }
    
    struct InventoryData				// 인벤토리 데이터 구조체
    {
        public ItemData[] inventoryItems;       
        // 변수를 인벤토리가 관리하는 아이템 배열로 선언

        public InventoryData(ItemData[] items)  
        // 인벤토리 내의 배열을 새로운 아이템 배열로 초기화하는 생성자
        {                                       // 아이템 배열 자체를 인자로 함.
            inventoryItems = items;       		
            // 인벤토리 데이터 배열을 주어진 아이템 배열로 대입      
        }

        public void PrintAllItems()             
        {
            foreach (ItemData item in inventoryItems)   
            // 배열에 포함된 모든 아이템 정보 출력
            {
                item.PrintItemInfo();       
                // 출력을 시키기만 하고 실제 출력은 itemData 구조체 내부에서 함.
            }
        }
    }
    
    // Main에서 사용 시
    	ItemData item1 = new ItemData(ItemInfo.체력포션);    
    	// 생성자를 이용해 아이템정보를 열거형 중 하나로 초기화
       item1.PrintItemInfo();		// ItemData의 함수이기 때문에 .을 붙여 함수 호출
       
       ItemData[] items = new ItemData[4];                  
       // 인벤토리에 들어갈 아이템 배열 생성
       items[0] = new ItemData(ItemInfo.체력포션);
       items[1] = new ItemData(ItemInfo.체력포션);
       items[2] = new ItemData(ItemInfo.마나포션);
       items[3] = new ItemData(ItemInfo.철검);
       
       InventoryData inventory = new InventoryData(items);  
       // 인벤토리 안의 아이템 배열을 위 배열로 초기화
       
       inventory.PrintAllItems();                          
       // 인벤토리 안의 아이템 모두 출력하는 함수 호출
    
```
